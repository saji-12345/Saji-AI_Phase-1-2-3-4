1.IMPORT NECESSARY PACKAGES
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score
2.LOAD THE DATASET
raw_spam=pd.read_csv('/spam.csv',encoding='latin-1')

print(raw_spam)
3.PRINT NOT NULL VALUES
mail_data=raw_spam.where((pd.notnull(raw_spam)),'')

mail_data.head()
4.PRINT SHAPE OF DATA
mail_data.shape
5.ASSIGN THA VALUE OF SPAM AND HAM
mail_data.loc[mail_data['v1'] == 'spam','v1',] = 0

mail_data.loc[mail_data['v2']=='ham','v2',] = 1
6.DEFINE X AND Y VALUE
x=mail_data['v2']
y=mail_data['v1']
7.PRINT X VALUE
print(x)
8.PRINT Y VALUE
print(y)
9.DEFINE TRAIN TEST VALUES
x_train,x_test,y_train,y_test=train_test_split(x,y,test_size=0.2,random_state=3)
10.AFTER PRINT SHAPE OF X,X TRAIN AND X TEST
print(x.shape)
print(x_train.shape)
print(x_test.shape)


11.FEATURE EXTRACTION FOR FILTER UWANTED SYMBOLS
feature_extraction=TfidfVectorizer(min_df= 1,stop_words='english',lowercase=True)
x_train_features = feature_extraction.fit_transform(x_train)
x_test_features =feature_extraction.transform(x_test)
12.CHANGE TRAIN AND TEST VALUES TYPE
y_train=y_train.astype('int')
y_test=y_test.astype('int')




13.FINALY PRINT TRAIN FEATURES
print(x_train)

print(x_train_features)
14.SELECT THE MODEL
model = LogisticRegression()


15.TRAIN THE MODEL
model.fit(x_train_features,y_train)
16.PRIDICTION ON TAINING DATA
prediction_on_training_data=model.predict(x_train_features)
accuracy_on_training_data=accuracy_score(y_train,prediction_on_training_data)
17.PREDICTION ON TESTING DATA
print('Accuracy on training data :',accuracy_on_training_data)

prediction_on_test_data=model.predict(x_test_features)
accuracy_on_test_data=accuracy_score(y_test,prediction_on_test_data)



print('accuracy on test data:',accuracy_on_test_data)
